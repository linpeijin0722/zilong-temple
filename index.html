// 處理網頁請求 function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("日期設定"); var range = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1); var dates = range.getValues().map(function(row) { return row[0]; }); return ContentService.createTextOutput(JSON.stringify({ "dates": dates })) .setMimeType(ContentService.MimeType.JSON); } // 處理報名提交 function doPost(e) { var params = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("工作表1"); if (!sheet) sheet = ss.getSheets()[0]; var data = sheet.getDataRange().getValues(); var name = params.name, phone = params.phone, reason = params.reason, bookDate = params.bookDate, lineId = params.lineId; var orderCount = 1; for (var i = 1; i < data.length; i++) { if (data[i][4] === bookDate) { orderCount++; } } var newRow = [new Date(), name, phone, reason, bookDate, orderCount, lineId]; sheet.appendRow(newRow); return ContentService.createTextOutput(JSON.stringify({ "status": "success", "orderNum": orderCount })) .setMimeType(ContentService.MimeType.JSON); }
